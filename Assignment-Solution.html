def getTopFiveRecommentations(predictions, n=5):
    top5 = defaultdict(list)
    for uid, iid, true_r, est, _ in predictions:
        top5[uid].append((iid, est))

    for uid, ratings in top5.items():
        ratings.sort(key=lambda x: x[1], reverse=True)
        top5[uid] = ratings[:n]

    return top5
